DROP TABLE IF EXISTS `order_info`;

CREATE TABLE `order_info` (
  `orderType` varchar(1) NOT NULL COMMENT '订单类型,A自主,B渠道,C主办方',
  `sourceOfOrder` varchar(100) NOT NULL COMMENT '下单来源',
  `orderid` varchar(100) NOT NULL COMMENT '内部订单号',
  `customorderid` varchar(100) DEFAULT NULL COMMENT '用户订单号',
  `dateoforder` datetime NOT NULL COMMENT '下单日期',
  `dateofissuance` datetime DEFAULT NULL COMMENT '发货日期,可为空',
  `statusoforder` int(1) NOT NULL COMMENT '订单状态 0=交易关闭 1=已完成',
  `methodofpayment` varchar(100) DEFAULT NULL COMMENT '支付方式,交易关闭时为空',
  `statusofpayment` int(1) NOT NULL COMMENT '支付状态 0=未支付 1=已支付',
  `methodOfDistribution` varchar(100) DEFAULT NULL COMMENT '配送方式',
  `consignee` varchar(100) DEFAULT NULL COMMENT '收货人,可为空',
  `phoneNumberOfConsignee` varchar(100) DEFAULT NULL COMMENT '收货人手机号,可为空',
  `phoneNumberOfCustom` varchar(100) DEFAULT NULL COMMENT '下单人手机号,可为空',
  `discountGrade` varchar(100) DEFAULT NULL COMMENT '下单会员等级折扣',
  `perksOfMembershi` varchar(100) DEFAULT NULL COMMENT '订单会员权益,可为空',
  `nameOfCommodity` varchar(100) NOT NULL COMMENT '商品名称',
  `performanceId` varchar(100) NOT NULL COMMENT '场次Id',
  `nameOfPerformance` varchar(100) NOT NULL COMMENT '场次名称',
  `ieldOfPerformance` varchar(100) DEFAULT NULL COMMENT '馆厅',
  `dateOfPerformance` datetime NOT NULL COMMENT '演出日期',
  `priceAndNumber` varchar(100) NOT NULL COMMENT '票价',
  `priceOfTicket` double NOT NULL COMMENT '单价',
  `numberOfTicket` int(11) NOT NULL COMMENT '张数',
  `shouldPayment` double DEFAULT NULL COMMENT '应收',
  `actualPayment` double NOT NULL COMMENT '实收=应收*折扣',
  `ticketPurpose` varchar(100) DEFAULT NULL COMMENT '出票用途,可以为空'
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
